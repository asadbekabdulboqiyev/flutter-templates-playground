$ flutter run

$ mkdir lib/pages lib/pages/auth lib/services lib/providers lib/widgets
$ touch lib/main.dart lib/pages/home_page.dart lib/pages/profile_page.dart
$ touch lib/pages/auth/login_page.dart lib/services/api_service.dart
$ touch lib/providers/theme_provider.dart lib/widgets/custom_button.dart

$ flutter build ios --release

$ flutter test
$ flutter analyze
